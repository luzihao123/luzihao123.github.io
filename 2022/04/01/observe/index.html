设计模式–发布订阅模式

概述

发布订阅模式又称为观察者模式，当一个对象状态改变时，所有依赖于它的对象都会得到通知。在javascript中，一般用事件模型来替代传统的发布-订阅模式

案例

楼盘即将推出新房，想购买的客户分别在售楼处登记了自己的电话，新楼盘推出时候售楼处会遍历记录的电话，依次发送信息给他们

自定义事件

售楼处案例可以写成以下最简单写法, 定义一个售楼处对象，里面有个一数组存放着客户的订阅信息，一个listen方法做订阅，一个trigger方法做发布，通过push加入订阅信息，触发时遍历所有订阅信息，把订阅的方法(fn函数)全部运行一遍

    var salesOffics = {}
    salesOffices.clientList = []
    saleOffices.listen = function (fn) {
        this.clientList.push(fn)
    }
    salesOffices.trigger = function () {
        for (var i=0,fn;fn = this.clientList[i];i++) {
            fn.apply(this, arguments)
        }
    }

测试

    salsesOffics.listen(function (price, squareMeter) {
        console.log('价格 = ', price)
        console.log('面积 = ', squareMeter)
    })
    salsesOffics.listen(function (price, squareMeter) {
        console.log('价格 = ', price)
        console.log('面积 = ', squareMeter)
    })
    salsesOffics.trigger(2000000, 88)
    salsesOffics.trigger(3000000, 120)

这个案例的问题是假如我只想订阅88平米的房子， 120平米房子信息变化时候也会收到信息，为了区分订阅事件的不同，可以加一个key，把clientList由于 数组变成对象 ，来区分订阅内容

    var salesOffics = {}
    salesOffices.clientList = {}
    salesOffices.listen = function (fn, key) {
        if (!salesOffices.clientList[key]) {
            this.clientList = []
        }
        this.client[key].push(fn)
    }
    salesOffices.trigger = function () {
        let key = Array.prototype.shift.call(arguments) // 取出第一个参数并将这个参数从argument移除
        let fns = this.clientList[key]
        if (!fns || !fns.length) {
            return false
        }
        fns.forEach(f => f.apply(this, arguments))
    }

这样改写后在使用listen时候多加一个参数

    salesOffics.listen((price) => console.log('价格=', price), '88')
    salesOffics.listen((price) => console.log('价格=', price), '120')

这样两次订阅分别订阅了不同的面积，只有所对应面积的信息触发会通知到订阅者

通用实现

把发布订阅功能提出出来，放在一个单独的对象中，就可以在不同情况下重用这段代码
顺便加上删除订阅的代码

    var event = {
        clientList: {},
        listen: function(key, fn) {
            if (!this.clientList[key]) {
                this.clientList[key] = []
            }
            this.clientList[key].push(fn)
        },
        trigger: function() {
            let key = Array.prototype.shift.call(arguments)
            let fns = this.clientList[key]
            if (!fns || fns.length === 0) return false
            fns.forEach(f =>f.apply(this, arguments))
        },
        remove: function(key, fn) {
            // key: 订阅类型  fn: 具体谁的订阅，如不传，删除此类型下全部订阅
            let fns = this.clientList[key]
            if (!fns) {
                return false
            }
            if (!fn) {
                fns.length = 0
            } else {
                for (let l = fns.length; l >=0; l--) {
                    var _fn = fns[l]
                    if (_fn === fn) {
                        fns.splice(l, 1)
                    }
                }
            }
        }
    }
    var installEvent = function(obj) {
        event.forEach(i => {
            obj[i] = event[i]
        })
    }

先发布后订阅

也许有时候需要先发布信息后进行订阅，我用了一个缓存对象来实现，在listen 中触发缓存，trigger中加入缓存

    var Event = {
        cache: {},
        clientList: {},
        listen: function(key, fn) {
            if (!this.clientList[key]) {
                this.clientList[key] = []
            }
            this.clientList[key].push(fn)
            if (this.cache[key]) {
                this.cache[key].forEach(v=>{
                   fn.apply(this, v);
                })
                this.cache[key] = []
            }
        },
        trigger: function() {
            let key = Array.prototype.shift.call(arguments)
            let fns = this.clientList[key]
            if (!fns || fns.length === 0) {
                if (!this.cache[key]) {
                    this.cache[key] = []
                }
                this.cache[key].push(arguments)
            } else {
                fns.forEach(f =>f.apply(this, arguments))
            }
            
        }
    }
